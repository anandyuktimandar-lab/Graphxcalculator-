    <script>
        // Global variables
        let equations = [];
        let graphData = {};
        let isDarkMode = false;
        let showGrid = true;
        let showAxisLabels = true;
        let zoomLevel = 1;
        let panOffset = { x: 0, y: 0 };
        const canvas = document.getElementById('graphCanvas');
        const ctx = canvas.getContext('2d');
        const equationsContainer = document.getElementById('equationsContainer');
        const equationInput = document.getElementById('equationInput');
        const addEquationBtn = document.getElementById('addEquationBtn');
        const darkModeToggle = document.getElementById('darkModeToggle');
        const settingsBtn = document.getElementById('settingsBtn');
        const resetBtn = document.getElementById('resetBtn');
        const exportBtn = document.getElementById('exportBtn');
        const settingsModal = document.getElementById('settingsModal');
        const closeSettings = document.getElementById('closeSettings');
        const saveSettings = document.getElementById('saveSettings');
        const darkModeSetting = document.getElementById('darkModeSetting');
        const gridLinesSetting = document.getElementById('gridLinesSetting');
        const axisLabelsSetting = document.getElementById('axisLabelsSetting');
        const zoomInBtn = document.getElementById('zoomInBtn');
        const zoomOutBtn = document.getElementById('zoomOutBtn');
        const panBtn = document.getElementById('panBtn');
        const hoverCoord = document.getElementById('hoverCoord');

        // Initialize
        function init() {
            resizeCanvas();
            setupEventListeners();
            drawGraph();
            addDefaultEquations();
        }

        // Set up event listeners
        function setupEventListeners() {
            window.addEventListener('resize', resizeCanvas);
            canvas.addEventListener('mousemove', handleMouseMove);
            canvas.addEventListener('mouseleave', hideHoverCoord);
            canvas.addEventListener('wheel', handleWheel);
            addEquationBtn.addEventListener('click', addEquation);
            equationInput.addEventListener('keypress', function(e) {
                if (e.key === 'Enter') addEquation();
            });
            darkModeToggle.addEventListener('click', toggleDarkMode);
            settingsBtn.addEventListener('click', openSettings);
            closeSettings.addEventListener('click', closeSettingsModal);
            saveSettings.addEventListener('click', saveSettingsHandler);
            resetBtn.addEventListener('click', resetGraph);
            exportBtn.addEventListener('click', exportGraph);
            
            // Settings toggles
            darkModeSetting.addEventListener('click', () => {
                darkModeSetting.classList.toggle('bg-blue-500');
                darkModeSetting.classList.toggle('bg-gray-300');
                isDarkMode = !isDarkMode;
                toggleDarkMode();
            });
            
            gridLinesSetting.addEventListener('click', () => {
                gridLinesSetting.classList.toggle('bg-blue-500');
                gridLinesSetting.classList.toggle('bg-gray-300');
                showGrid = !showGrid;
                drawGraph();
            });
            
            axisLabelsSetting.addEventListener('click', () => {
                axisLabelsSetting.classList.toggle('bg-blue-500');
                axisLabelsSetting.classList.toggle('bg-gray-300');
                showAxisLabels = !showAxisLabels;
                drawGraph();
            });

            // Zoom buttons
            zoomInBtn.addEventListener('click', () => {
                zoomLevel *= 1.2;
                drawGraph();
            });
            
            zoomOutBtn.addEventListener('click', () => {
                zoomLevel /= 1.2;
                drawGraph();
            });
            
            panBtn.addEventListener('click', () => {
                // Toggle panning mode
            });
        }

        // Resize canvas to fit container
        function resizeCanvas() {
            const container = canvas.parentElement;
            canvas.width = container.clientWidth;
            canvas.height = container.clientHeight;
            drawGraph();
        }

        // Add default equations for demo
        function addDefaultEquations() {
            addEquationFromInput('y = x^2');
            addEquationFromInput('y = sin(x)');
            addEquationFromInput('y = 2*x + 3');
        }

        // Add equation from input field
        function addEquation() {
            const equationText = equationInput.value.trim();
            if (equationText) {
                addEquationFromInput(equationText);
                equationInput.value = '';
            }
        }

        // Add equation from text
        function addEquationFromInput(equationText) {
            try {
                const equation = {
                    id: Date.now(),
                    expression: equationText,
                    color: getRandomColor(),
                    visible: true
                };
                equations.push(equation);
                renderEquations();
                drawGraph();
            } catch (error) {
                alert('Invalid equation format. Please check syntax.');
            }
        }

        // Render equation list
        function renderEquations() {
            equationsContainer.innerHTML = '';
            equations.forEach(eq => {
                const equationItem = document.createElement('div');
                equationItem.className = 'equation-item flex justify-between items-center bg-gray-50 p-2 rounded dark:bg-gray-700';
                equationItem.innerHTML = `
                    <div class="flex items-center">
                        <div class="w-4 h-4 rounded-full mr-2" style="background-color: ${eq.color}"></div>
                        <span class="font-mono text-sm">${eq.expression}</span>
                    </div>
                    <div class="flex space-x-1">
                        <button onclick="toggleEquation('${eq.id}')" class="text-xs px-2 py-1 bg-gray-200 rounded hover:bg-gray-300 dark:bg-gray-600 dark:hover:bg-gray-500">
                            ${eq.visible ? 'Hide' : 'Show'}
                        </button>
                        <button onclick="deleteEquation('${eq.id}')" class="text-xs px-2 py-1 bg-red-500 text-white rounded hover:bg-red-600">
                            Delete
                        </button>
                    </div>
                `;
                equationsContainer.appendChild(equationItem);
            });
        }

        // Delete equation
        function deleteEquation(id) {
            equations = equations.filter(eq => eq.id !== id);
            renderEquations();
            drawGraph();
        }

        // Toggle equation visibility
        function toggleEquation(id) {
            const equation = equations.find(eq => eq.id === id);
            if (equation) {
                equation.visible = !equation.visible;
                drawGraph();
            }
        }

        // Generate random color
        function getRandomColor() {
            const colors = ['#3b82f6', '#10b981', '#8b5cf6', '#f59e0b', '#ef4444', '#ec4899'];
            return colors[Math.floor(Math.random() * colors.length)];
        }

        // Handle mouse move for hover coordinates
        function handleMouseMove(e) {
            const rect = canvas.getBoundingClientRect();
            const x = e.clientX - rect.left;
            const y = e.clientY - rect.top;
            
            const coord = screenToGraphCoords(x, y);
            showHoverCoord(coord.x.toFixed(2), coord.y.toFixed(2), x, y);
        }

        // Show hover coordinates
        function showHoverCoord(x, y, screenX, screenY) {
            hoverCoord.textContent = `(${x}, ${y})`;
            hoverCoord.style.left = `${screenX + 10}px`;
            hoverCoord.style.top = `${screenY - 20}px`;
            hoverCoord.classList.remove('hidden');
        }

        // Hide hover coordinates
        function hideHoverCoord() {
            hoverCoord.classList.add('hidden');
        }

        // Handle mouse wheel for zooming
        function handleWheel(e) {
            e.preventDefault();
            const zoomIntensity = 0.1;
            if (e.deltaY < 0) {
                zoomLevel *= (1 + zoomIntensity);
            } else {
                zoomLevel *= (1 - zoomIntensity);
            }
            zoomLevel = Math.max(0.1, Math.min(zoomLevel, 10));
            drawGraph();
        }

        // Convert screen coordinates to graph coordinates
        function screenToGraphCoords(screenX, screenY) {
            const width = canvas.width;
            const height = canvas.height;
            
            // Center of canvas
            const centerX = width / 2;
            const centerY = height / 2;
            
            // Scale factor (graph units per pixel)
            const scale = 20 * zoomLevel;
            
            // Convert to graph coordinates
            const x = (screenX - centerX - panOffset.x) / scale;
            const y = -(screenY - centerY - panOffset.y) / scale;
            
            return { x, y };
        }

        // Draw the graph
        function drawGraph() {
            const width = canvas.width;
            const height = canvas.height;
            
            // Clear canvas
            ctx.clearRect(0, 0, width, height);
            
            // Draw background
            ctx.fillStyle = isDarkMode ? '#0f172a' : '#f8fafc';
            ctx.fillRect(0, 0, width, height);
            
            // Draw grid
            if (showGrid) {
                drawGrid(width, height);
            }
            
            // Draw axes
            drawAxes(width, height);
            
            // Plot equations
            equations.forEach(eq => {
                if (eq.visible) {
                    plotEquation(eq.expression, eq.color);
                }
            });
        }

        // Draw grid lines
        function drawGrid(width, height) {
            const scale = 20 * zoomLevel;
            
            // Vertical grid lines
            for (let x = 0; x <= width; x += scale) {
                ctx.beginPath();
                ctx.moveTo(x, 0);
                ctx.lineTo(x, height);
                ctx.strokeStyle = isDarkMode ? 'rgba(255, 255, 255, 0.05)' : 'rgba(0, 0, 0, 0.1)';
                ctx.lineWidth = 1;
                ctx.stroke();
            }
            
            // Horizontal grid lines
            for (let y = 0; y <= height; y += scale) {
                ctx.beginPath();
                ctx.moveTo(0, y);
                ctx.lineTo(width, y);
                ctx.strokeStyle = isDarkMode ? 'rgba(255, 255, 255, 0.05)' : 'rgba(0, 0, 0, 0.1)';
                ctx.lineWidth = 1;
                ctx.stroke();
            }
        }

        // Draw axes
        function drawAxes(width, height) {
            const centerX = width / 2 + panOffset.x;
            const centerY = height / 2 + panOffset.y;
            const scale = 20 * zoomLevel;
            
            // X-axis
            ctx.beginPath();
            ctx.moveTo(0, centerY);
            ctx.lineTo(width, centerY);
            ctx.strokeStyle = isDarkMode ? '#94a3b8' : '#333';
            ctx.lineWidth = 2;
            ctx.stroke();
            
            // Y-axis
            ctx.beginPath();
            ctx.moveTo(centerX, 0);
            ctx.lineTo(centerX, height);
            ctx.strokeStyle = isDarkMode ? '#94a3b8' : '#333';
            ctx.lineWidth = 2;
            ctx.stroke();
            
            // Draw axis labels
            if (showAxisLabels) {
                ctx.fillStyle = isDarkMode ? '#cbd5e1' : '#333';
                ctx.font = '12px Arial';
                ctx.textAlign = 'center';
                
                // X-axis labels
                for (let i = -10; i <= 10; i++) {
                    if (i === 0) continue;
                    const x = centerX + i * scale;
                    if (x >= 0 && x <= width) {
                        ctx.fillText(i.toString(), x, centerY + 15);
                    }
                }
                
                // Y-axis labels
                ctx.textAlign = 'right';
                for (let i = -10; i <= 10; i++) {
                    if (i === 0) continue;
                    const y = centerY - i * scale;
                    if (y >= 0 && y <= height) {
                        ctx.fillText(i.toString(), centerX - 5, y + 4);
                    }
                }
            }
        }

        // Plot equation
        function plotEquation(expression, color) {
            const width = canvas.width;
            const height = canvas.height;
            const centerX = width / 2 + panOffset.x;
            const centerY = height / 2 + panOffset.y;
            const scale = 20 * zoomLevel;
            
            ctx.beginPath();
            ctx.strokeStyle = color;
            ctx.lineWidth = 2;
            
            let firstPoint = true;
            
            // Evaluate the function over a range of x values
            for (let i = -20; i <= 20; i += 0.1) {
                const x = i;
                try {
                    // Parse and evaluate the expression with x substituted
                    const expr = math.parse(expression.replace(/y\s*=\s*/g, ''));
                    const scope = { x: x };
                    const result = expr.evaluate(scope);
                    
                    // Convert to screen coordinates
                    const screenX = centerX + x * scale;
                    const screenY = centerY - result * scale;
                    
                    if (firstPoint) {
                        ctx.moveTo(screenX, screenY);
                        firstPoint = false;
                    } else {
                        ctx.lineTo(screenX, screenY);
                    }
                } catch (error) {
                    // Skip invalid points
                }
            }
            
            ctx.stroke();
        }

        // Toggle dark mode
        function toggleDarkMode() {
            document.body.classList.toggle('dark:bg-darkBackground');
            document.body.classList.toggle('dark:text-darkText');
            document.body.classList.toggle('bg-gray-100');
            document.body.classList.toggle('text-gray-900');
            drawGraph();
        }

        // Open settings modal
        function openSettings() {
            settingsModal.classList.remove('hidden');
        }

        // Close settings modal
        function closeSettingsModal() {
            settingsModal.classList.add('hidden');
        }

        // Save settings handler
        function saveSettingsHandler() {
            closeSettingsModal();
            drawGraph();
        }

        // Reset graph
        function resetGraph() {
            equations = [];
            renderEquations();
            zoomLevel = 1;
            panOffset = { x: 0, y: 0 };
            drawGraph();
        }

        // Export graph
        function exportGraph() {
            const link = document.createElement('a');
            link.download = 'math-graph.png';
            link.href = canvas.toDataURL();
            link.click();
        }

        // Initialize the application
        window.onload = init;
    </script>
